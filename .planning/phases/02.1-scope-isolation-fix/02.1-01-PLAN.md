import asyncio, duckdb
from prme.storage.schema import initialize_database
from prme.storage.event_store import EventStore
from prme.models.events import Event
from prme.types import Scope

conn = duckdb.connect(':memory:')
initialize_database(conn)
store = EventStore(conn)

async def test():
    # Test 1: Store event with default scope (PERSONAL)
    e1 = Event(content='personal note', user_id='u1', role='user')
    await store.append(e1)
    retrieved = await store.get(str(e1.id))
    assert retrieved is not None
    assert retrieved.scope == Scope.PERSONAL, f'Expected PERSONAL, got {retrieved.scope}'

    # Test 2: Store event with PROJECT scope
    e2 = Event(content='project note', user_id='u1', role='user', scope=Scope.PROJECT)

    assert retrieved2.scope == Scope.PROJECT, f'Expected PROJECT, got {retrieved2.scope}'

    # Test 3: Multi-scope query
    all_events = await store.get_by_user('u1')

    personal_only = await store.get_by_user('u1', scopes=[Scope.PERSONAL])
    assert len(personal_only) == 1
    assert personal_only[0].scope == Scope.PERSONAL

    project_only = await store.get_by_user('u1', scopes=[Scope.PROJECT])
    assert len(project_only) == 1
    assert project_only[0].scope == Scope.PROJECT

    multi = await store.get_by_user('u1', scopes=[Scope.PERSONAL, Scope.PROJECT])
    assert len(multi) == 2

    print('All EventStore tests passed')

asyncio.run(test())
