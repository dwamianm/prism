import duckdb
from prme.storage.schema import initialize_database
from prme.storage.duckpgq_graph import DuckPGQGraphStore

async def test():
    conn = duckdb.connect(':memory:')
    initialize_database(conn)
    gs = DuckPGQGraphStore(conn)
    merger = EntityMerger(gs)

    # First call: creates new entity

    conn.close()

import asyncio
import duckdb
from prme.storage.schema import initialize_database
from prme.storage.duckpgq_graph import DuckPGQGraphStore
from prme.ingestion.supersedence import SupersedenceDetector, _predicates_match
from prme.models.nodes import MemoryNode
from prme.models.edges import MemoryEdge
from prme.types import NodeType, EdgeType, Scope, LifecycleState

async def test():
    conn = duckdb.connect(':memory:')
    initialize_database(conn)
    gs = DuckPGQGraphStore(conn)
    detector = SupersedenceDetector(gs)

    # Create entity
    entity = MemoryNode(user_id='u1', node_type=NodeType.ENTITY, content='Sarah', scope=Scope.PERSONAL, metadata={'entity_type': 'person'})
    entity_id = await gs.create_node(entity)

    # Create existing fact: Sarah works_at Google
    old_fact = MemoryNode(user_id='u1', node_type=NodeType.FACT, content='Sarah works at Google', scope=Scope.PERSONAL, metadata={'subject': 'Sarah', 'predicate': 'works_at', 'object': 'Google'})
    old_fact_id = await gs.create_node(old_fact)

    # Link entity to fact
    edge = MemoryEdge(source_id=entity.id, target_id=old_fact.id, edge_type=EdgeType.HAS_FACT, user_id='u1')
    await gs.create_edge(edge)

    # Create new contradicting fact: Sarah works_at Meta
    new_fact = MemoryNode(user_id='u1', node_type=NodeType.FACT, content='Sarah works at Meta', scope=Scope.PERSONAL, metadata={'subject': 'Sarah', 'predicate': 'works_at', 'object': 'Meta'})
    new_fact_id = await gs.create_node(new_fact)

    # Detect supersedence
    superseded = await detector.detect_and_supersede(new_fact_id, entity_id, 'works_at', 'Meta', 'u1')
    print(f'Superseded count: {len(superseded)} (expected 1)')
    print(f'Superseded old fact: {old_fact_id in superseded}')

    # Predicate matching tests
    print(f'Exact match: {_predicates_match(\"works_at\", \"works_at\")}')
    print(f'Equiv match: {_predicates_match(\"works_at\", \"employed_by\")}')
    print(f'No match: {not _predicates_match(\"works_at\", \"lives_in\")}')

    conn.close()

asyncio.run(test())
"`
    Expected: Superseded count: 1, Superseded old fact: True, Exact match: True, Equiv match: True, No match: True
  </verify>
